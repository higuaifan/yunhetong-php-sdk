/**
 * 用户的实体类
 * Created by PhpStorm.
 * User: SeanWu
 * Date: 2016/5/23
 * Time: 19:13
 */
class User
{
    public $appId;    // 应用Id
    public $appUserId; // 用户在第三方应用的唯一标识
    public $userType; // 用户类型,1是个人，2是企业
    public $cellNum; //  电话号码，为1开头的11为数字
    public $userName;// 用户名称
    public $certifyType; // 实名认证类型1身份证2护照3军官证4营业执照5组织机构代码证
    public $certifyNumber; // 用户实名认证时候的证件号码，可以是对应的身份证、营业执照、组织机构代码证或者其他证件号码，原则上不能大于 30 个字符
    public $createSignature = 0;  // 是否自动创建签名，在导入用户并且当值为 1 时，会为导入的用户自动创建签名，0的话就不会，这个值只在用户第一次导入时有效
}
