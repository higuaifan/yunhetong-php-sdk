        $this->noticeType = $jo["noticeType"];
        if ($this->noticeType == 2) {
            $this->signDigest = $jo["signDigest"];

        print_r($jo);

                // "用户:" + __key + ", 已签署";

                $this->noSignUsers[] = $key;
            }

        }


    }


}
